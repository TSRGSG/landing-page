// Project: Interactive To-Do List
class ToDoList {
  constructor() {
    this.tasks = [];
  }

  addTask(task) {
    this.tasks.push({ task, completed: false });
  }

  completeTask(taskIndex) {
    if (taskIndex >= 0 && taskIndex < this.tasks.length) {
      this.tasks[taskIndex].completed = true;
    }
  }

  displayTasks() {
    console.log("Tasks:");
    this.tasks.forEach((item, index) => {
      const status = item.completed ? "[✔]" : "[ ]";
      console.log(`${index + 1}. ${status} ${item.task}`);
    });
  }
}

// Usage
const myToDoList = new ToDoList();
myToDoList.addTask("Write my portfolio website code (ポートフォリオ Web サイトのコードを書きます。)");
myToDoList.addTask("Deploy on cloud platform (クラウドプラットフォーム上に展開します。)");
myToDoList.addTask("Translate to Japanese (日本語に翻訳します。)");

myToDoList.completeTask(1);

// Display Tasks
myToDoList.displayTasks();
